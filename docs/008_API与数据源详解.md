### 3.3 从新浪获取LOF基金历史收盘价格数据

**函数?*：`fetch_lof_price_data`

```python
def fetch_lof_price_data(self, fund_code, existing_data=None):
    """从新浪获取LOF基金历史收盘价格数据
    
    Args:
        fund_code: LOF基金代码
        existing_data: 现有的数据，用于棢查是否需要爬?
    
    Returns:
        DataFrame: 包含LOF基金历史数据的DataFrame
    """
    logger.info(f"从新浪获取LOF基金 {fund_code} 历史收盘价格数据")
    
    # 棢查是否需要爬?
    if existing_data is not None:
        # 棢查是否有朢新数?
        today = datetime.now().date().strftime('%Y-%m-%d')
        if not existing_data.empty:
            latest_date = existing_data['日期'].max()
            if latest_date == today:
                logger.info(f"已有朢新数据，跳过爬取")
                return existing_data
    
    # 计算日期范围
    end_date = datetime.now().date()
    start_date = (datetime.now() - timedelta(days=30)).date()
    days_to_fetch = (end_date - start_date).days + 1
    days_to_fetch = max(days_to_fetch, 30)
    
    # 根据基金代码前缀确定交易扢?0弢头为上交扢?6弢头为深交扢?
    exchange_prefix = 'sh' if fund_code.startswith('50') else 'sz'
    sina_url = f"http://money.finance.sina.com.cn/quotes_service/api/json_v2.php/CN_MarketData.getKLineData?symbol={exchange_prefix}{fund_code}&scale=240&ma=no&datalen={days_to_fetch}"
    logger.info(f"新浪API获取 {days_to_fetch} 天的数据")
    
    # 发请求，添加重试机制
    max_retries = 3
    retry_count = 0
    
    sina_data = None
    while retry_count < max_retries:
        try:
            response = requests.get(sina_url, headers=self.headers, timeout=15, verify=False)
            logger.info(f"新浪API响应状码: {response.status_code}")
            response.raise_for_status()  # 棢查请求是否成?
            
            # 解析响应
            import json
            sina_data = response.json()
            break  # 请求成功，出重试循?
        except (requests.exceptions.ConnectionError, requests.exceptions.Timeout) as e:
            retry_count += 1
            logger.error(f"获取新浪历史数据失败，正在重?({retry_count}/{max_retries}): {e}")
            time.sleep(3)  # 等待3秒后重试
        except Exception as e:
            logger.error(f"获取新浪历史数据失败: {e}")
            # 继续重试
            retry_count += 1
            if retry_count < max_retries:
                logger.info(f"正在重试 ({retry_count}/{max_retries})")
                time.sleep(3)
            else:
                # 扢有重试失败，无法获取数据
                logger.error("扢有重试失败，无法获取新浪的LOF历史收盘价格数据")
                # 即使新浪失败，也继续处理，使用现有数?
                sina_data = []
    
    if isinstance(sina_data, list):
        logger.info(f"找到 {len(sina_data)} 条新浪历史交易记?)
        
        # 解析新浪历史数据
        lof_data = []
        date_set = set()  # 用于跟踪已处理的日期
        
        # 获取今天的日期字符串
        today_str = datetime.now().date().strftime('%Y-%m-%d')
        
        for item in sina_data:
            date = item.get('day')  # 日期
            close = item.get('close')  # 收盘?
            
            if not date or not close:
                continue
            
            # 棢查是否为今天的数据，如果是，根据时间决定是否处理
            if date == today_str:
                # 获取当前时间
                now = datetime.now()
                # 如果当前时间超过15:01，处理今天的数据作为收盘?
                if now.hour > 15 or (now.hour == 15 and now.minute >= 1):
                    logger.info(f"处理今天({today_str})的收盘价数据")
                else:
                    logger.info(f"跳过今天({today_str})的数据，?2模块处理")
                    continue
            
            # 跳过已经处理过的日期
            if date in date_set:
                continue
            date_set.add(date)
            
            try:
                close = float(close)  # 收盘价（历史数据?
            except ValueError:
                continue
            
            lof_data.append({
                '日期': date,
                'LOF交易价格': close  # 历史收盘?
            })
        
        if lof_data:
            # 转换为DataFrame
            df = pd.DataFrame(lof_data)
            # 按日期排序（降序?
            df['日期'] = pd.to_datetime(df['日期'], format='%Y-%m-%d')
            df = df.sort_values('日期', ascending=False)
            # 保留完整的日期格式，包含年份
            df['日期'] = df['日期'].dt.strftime('%Y-%m-%d')
            
            logger.info(f"成功获取LOF基金 {fund_code} 历史收盘价格数据，共{len(df)}条记?)
            return df
        else:
            logger.warning("新浪返回的数据为?)
            return None
    else:
        logger.error("新浪返回的数据格式不正确")
        return None
```

```python
    }
    
    for idx in indices:
        if idx in index_url_map:
            url = index_url_map[idx]
            
            try:
                response = requests.get(url, headers=sina_headers, timeout=15, verify=False)
                response.encoding = 'utf-8'
                if response.status_code == 200:
                    data_str = response.text
                    if data_str:
                        # 使用正则表达式提取指数价?
                        import re
                        # 匹配类似 "6616.85(0.08%)" 的模?
                        match = re.search(r'"([\d,]+\.\d+)\([^)]+\)"', data_str)
                        if match:
                            # 移除逗号，转换为浮点?
                            close_price_str = match.group(1).replace(',', '')
                            close_price = float(close_price_str)
                            index_data[idx] = close_price
                            logger.info(f"{idx} 指数价格: {close_price}")
                else:
                    logger.error(f"请求 {idx} 指数数据失败，状态码: {response.status_code}")
            except Exception as e:
                logger.error(f"获取 {idx} 指数数据失败: {e}")
    
    return index_data
```

### 3.7 从交易所 API 获取基金场内份额 (JSL 与看板核心升级)

**类名**：`SZSEFundSharesFetcher` / `DataFetcher` 中集成的场内份额获取逻辑

**功能**：系统通过直接请求上交所（SSE）与深交所（SZSE）官方 API，获取全市场 LOF 基金的场内份额与流通规模。

#### 1) 上交所批量获取与缓存机制 (SSE)
针对上交所基金（代码 `50` 或 `51` 开头），系统支持**一键拉取全市场数据**并本地缓存。
* **API 地址**：`https://query.sse.com.cn/commonQuery.do?isPagination=true&sqlId=COMMON_SSE_SJ_JJSJ_JJGM_LOFGMTJ_L&PRODUCT_TYPE=11,14,15&SEARCH_DATE=&type=inParams&pageHelp.pageSize=10000`
* **防爬限制**：请求头中必须附带 `Referer: https://www.sse.com.cn/`。
* **主要方法**：`_fetch_sse_shares_all(self)` 会在首次遇到沪市基金时拉取全市场份额并缓存在 `_sse_cache` 中，避免频繁请求被封锁。

#### 2) 深交所逐个获取机制 (SZSE)
针对深交所基金（代码 `15`/`16` 开头），由于深交所无公开全市场大一统接口，且反爬机制极其严苛（高频并发会触发 IP 熔断），系统使用**延迟单只抓取或通过云端 VPS 采集同步**。
* **API 地址**：`https://www.szse.cn/api/report/ShowReport/data?SHOWTYPE=JSON&CATALOGID=1945_LOF&txtQueryKeyAndJC={fund_code}`
* **主要方法**：`_fetch_szse_fund_by_code(self, fund_code)`
* **防封技术**：
  - 限制 9:15 之前不请求（数据未更新无需请求）。
  - 设置 20 分钟缓存机制。
  - 请求之间随机休眠 2.0 至 4.0 秒。
  - 触发全局熔断机制（一旦拦截，后续所有深市基金不再请求深交所 API，保证系统稳定）。

#### 3) 实现代码参考

```python
def fetch_shares_safe(self, fund_code):
```
